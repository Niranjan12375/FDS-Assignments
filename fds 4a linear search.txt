def accept_array(A): #accepting the values from user
 n = int(input("Enter the total number of students : "))
 for i in range(n):
 x = int(input("Enter the roll number of student %d : " %(i+1)))
 A.append(x)
 print("Student Information Accept Successfully\n\n")
 return n
def display_array(A,n): #displaying the enetered values
 if(n==0):
 print("\nNo records in the database")
 else:
 print("Students Array : ", end=' ')
 for i in range(n):
 print("%d "%A[i],end=' ')
 print("\n");


def Linear_Search(A,n,X):#normal search
 for i in range(n):
 if(A[i]==X):
 return i
 return -1
def Sentinental_Search(A,n,X):#works same with different logic
 last = A[n-1]
 i=0
 A[n-1]=X
 while(A[i] != X):
 i=i+1
 A[n-1]=last
 if((i<n-1) or (X==A[n-1])):
 return i
 else:
 return -1
